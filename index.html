6. Print the Pattern.

const Print_pattern = (N) => 
  {
    let star='';
   for(let i=1; i<=N; i++){
     for(let j=1;j<=i;j++)
     {
       star=star+'*';
     }
     star=star+"\n";
   }
   
   console.log(star);
  };
